# heatmap: top20 up/down by score
plot_top_score_heatmap <- function(expr, clin, de, samples, ha, title_text) {
  de <- add_score(de)

  de_sub <- de[match(rownames(expr), de$Gene), ]
  rownames(de_sub) <- de_sub$Gene

  up   <- de_sub[order(de_sub$score, decreasing = TRUE), , drop = FALSE]
  down <- de_sub[order(de_sub$score, decreasing = FALSE), , drop = FALSE]

  top_up   <- rownames(up)[1:20]
  top_down <- rownames(down)[1:20]

  genes_use <- c(top_up, top_down)

  mat <- as.matrix(expr[genes_use, samples, drop = FALSE])

  # z-score per protein (row), same as your heatmaps
  matz <- t(scale(t(mat)))
  matz[is.nan(matz)] <- 0

  hm_cols <- colorRamp2(c(-2, 0, 2), c("#2166AC", "white", "#B2182B"))

  Heatmap(
    matz,
    name = "Z-score",
    col = hm_cols,
    top_annotation = ha,
    cluster_rows = FALSE,
    cluster_columns = TRUE,
    show_row_names = TRUE,
    row_names_gp = gpar(fontsize = 13),
    show_column_names = TRUE,
    column_names_gp = gpar(fontsize = 8),
    row_title = title_text,
    column_title = NULL,
    height = unit(nrow(matz) * 5, "mm")
  ) |>
    draw(heatmap_legend_side = "right", annotation_legend_side = "right")
}

  y_test  <- ifelse(as.numeric(clin_test$TAPSE_sPAP)  < cutoff, 1, 0)

  # --- X matrices: samples x proteins ---
  Xtr_prot <- t(as.matrix(expr_train[, clin_train$Barcode, drop = FALSE]))
  Xte_prot <- t(as.matrix(expr_test[,  clin_test$Barcode,  drop = FALSE]))

  if (!use_covariates) {
    Xtr <- Xtr_prot
    Xte <- Xte_prot
    pf  <- rep(1, ncol(Xtr))
  } else {
    cov_tr <- data.frame(
      Age = as.numeric(clin_train$Age),
      Sex = factor(clin_train$Sex)
    )
    cov_te <- data.frame(
      Age = as.numeric(clin_test$Age),
      Sex = factor(clin_test$Sex, levels = levels(cov_tr$Sex))
    )

    Xtr_cov <- model.matrix(~ Age + Sex, data = cov_tr)[, -1, drop = FALSE]
    Xte_cov <- model.matrix(~ Age + Sex, data = cov_te)[, -1, drop = FALSE]

    Xtr <- cbind(Xtr_cov, Xtr_prot)
    Xte <- cbind(Xte_cov, Xte_prot)

    # Age/Sex unpenalized, proteins penalized
    pf <- c(rep(0, ncol(Xtr_cov)), rep(1, ncol(Xtr_prot)))
  }

  # --- CV LASSO ---
  set.seed(1)
  cvfit <- cv.glmnet(
    x = Xtr, y = y_train,
    family = "binomial",
    alpha = alpha,
    penalty.factor = pf,
    standardize = TRUE,
    nfolds = 3,
    type.measure = "auc"
  )

  # --- choose lambda giving 3–10 proteins (closest if none) ---
  lambdas <- cvfit$glmnet.fit$lambda
  beta_mat <- as.matrix(coef(cvfit$glmnet.fit, s = lambdas))  # rows = features+intercept, cols = lambdas

  # count proteins only
  feat_names <- rownames(beta_mat)
  prot_idx <- if (!use_covariates) {
    which(feat_names != "(Intercept)")
  } else {
    which(feat_names != "(Intercept)" & !(feat_names %in% colnames(Xtr)[1:(ncol(Xtr) - ncol(Xtr_prot))]))
    # easier: proteins are the columns of Xtr_prot
    which(feat_names %in% colnames(Xtr_prot))
  }

  nprot <- apply(beta_mat[prot_idx, , drop = FALSE] != 0, 2, sum)

  ok <- which(nprot >= 3 & nprot <= 10)
  if (length(ok) > 0) {
    # pick best CV AUC among those (closest to min cvm)
    j <- ok[which.min(cvfit$cvm[match(lambdas[ok], cvfit$lambda)])]
    lambda_use <- lambdas[j]
  } else {
    # pick lambda with nprot closest to 5
    j <- which.min(abs(nprot - 5))
    lambda_use <- lambdas[j]
  }

  # --- final coefficients at chosen lambda ---
  beta <- as.matrix(coef(cvfit, s = lambda_use))
  beta <- beta[beta[,1] != 0, , drop = FALSE]

  # proteins selected (exclude intercept + covariates)
  selected <- rownames(beta)
  selected <- selected[selected != "(Intercept)"]
  if (use_covariates) {
    selected_proteins <- selected[selected %in% colnames(Xtr_prot)]
  } else {
    selected_proteins <- selected
  }

  # --- predict in Poland using locked Zurich model ---
  p_train <- as.numeric(predict(cvfit, newx = Xtr, s = lambda_use, type = "response"))
  p_test  <- as.numeric(predict(cvfit, newx = Xte, s = lambda_use, type = "response"))

  # AUCs
  auc_train <- as.numeric(pROC::auc(pROC::roc(y_train, p_train, quiet = TRUE)))
  auc_test  <- as.numeric(pROC::auc(pROC::roc(y_test,  p_test,  quiet = TRUE)))

  list(
    cvfit = cvfit,
    lambda_use = lambda_use,
    selected_proteins = selected_proteins,
    beta_nonzero = beta,
    train = data.frame(Barcode = clin_train$Barcode, y = y_train, p = p_train),
    test  = data.frame(Barcode = clin_test$Barcode,  y = y_test,  p = p_test),
    auc_train = auc_train,
    auc_test = auc_test
  )
}

compute_panel_score <- function(expr, clin, genes, cutoff = 0.32) {

  # outcome
  y <- ifelse(as.numeric(clin$TAPSE_sPAP) < cutoff, 1, 0)

  # matrix samples x genes
  X <- t(as.matrix(expr[genes, clin$Barcode, drop = FALSE]))

  # z-score per protein within cohort
  Xz <- scale(X)

  # direction from Zurich logFC
  directions <- sign(de_z$logFC[match(genes, de_z$Gene)])

  # additive signed score
  score <- as.numeric(Xz %*% directions)

  data.frame(
    Barcode = clin$Barcode,
    y = y,
    score = score
  )
}

Chunk 2 — stability-selection bootstrap LASSO (Age/Sex unpenalized)

This runs B=500 bootstraps in Poland only and returns: - selection frequency - sign stability - coefficient summaries - OOB AUC per bootstrap (so you get a “typical performance” distribution without using Zurich)

library(Matrix)

make_design <- function(expr_mat, clin_df, include_age = TRUE, include_sex = TRUE){
  # expr_mat: proteins x samples, columns match clin_df$Barcode (in same order)
  X_prot <- t(expr_mat[, clin_df$Barcode, drop = FALSE])  # samples x proteins

  cov_parts <- list()
  if (include_age) {
    cov_parts$Age <- as.numeric(clin_df$Age)
  }
  if (include_sex) {
    cov_parts$Sex <- factor(clin_df$Sex)
  }

  if (length(cov_parts) == 0) {
    X <- X_prot
    pf <- rep(1, ncol(X))
    cov_names <- character(0)
  } else {
    cov_df <- as.data.frame(cov_parts)
    # model.matrix handles Sex dummy coding automatically
    X_cov <- model.matrix(~ ., data = cov_df)[, -1, drop = FALSE]  # remove intercept
    X <- cbind(X_cov, X_prot)

    # unpenalize covariates, penalize proteins
    pf <- c(rep(0, ncol(X_cov)), rep(1, ncol(X_prot)))
    cov_names <- colnames(X_cov)
  }

  list(X = X, pf = pf, cov_names = cov_names, prot_names = colnames(X_prot))
}

choose_lambda_by_protein_count <- function(cvfit, prot_names, cov_names,
                                          min_prot = 3, max_prot = 15){
  # choose a lambda that yields between min_prot and max_prot proteins (closest if none)
  lambdas <- cvfit$glmnet.fit$lambda
  beta_mat <- as.matrix(coef(cvfit$glmnet.fit, s = lambdas))

  feat_names <- rownames(beta_mat)
  prot_idx <- which(feat_names %in% prot_names)

  nprot <- apply(beta_mat[prot_idx, , drop = FALSE] != 0, 2, sum)

  ok <- which(nprot >= min_prot & nprot <= max_prot)
  if (length(ok) > 0) {
    # among ok, pick lambda with best CV AUC (min cvm because cv.glmnet stores -AUC? actually for AUC it stores 1-AUC)
    # safer: pick ok lambda that minimizes cvfit$cvm
    j <- ok[which.min(cvfit$cvm[match(lambdas[ok], cvfit$lambda)])]
    return(lambdas[j])
  }

  # if none in range, choose the lambda whose protein count is closest to target mid
  target <- round((min_prot + max_prot)/2)
  j <- which.min(abs(nprot - target))
  lambdas[j]
}

bootstrap_stability_lasso <- function(expr_mat, clin_df, y,
                                     B = 500,
                                     include_age = TRUE,
                                     include_sex = TRUE,
                                     alpha = 1,
                                     nfolds = 5,
                                     min_prot = 3,
                                     max_prot = 15,
                                     seed = 1){
  set.seed(seed)

  # Ensure order consistency
  stopifnot(all(clin_df$Barcode %in% colnames(expr_mat)))
  expr_mat <- expr_mat[, clin_df$Barcode, drop = FALSE]
  y <- as.numeric(y)

  des <- make_design(expr_mat, clin_df, include_age, include_sex)
  X_full <- des$X
  pf <- des$pf
  prot_names <- des$prot_names
  cov_names  <- des$cov_names

  n <- nrow(X_full)
  all_prots <- prot_names

  # storage
  sel_list <- vector("list", B)
  coef_list <- vector("list", B)
  auc_oob <- rep(NA_real_, B)
  lambda_used <- rep(NA_real_, B)

  for (b in seq_len(B)) {
    idx_boot <- sample(seq_len(n), size = n, replace = TRUE)
    idx_oob  <- setdiff(seq_len(n), unique(idx_boot))

    Xb <- X_full[idx_boot, , drop = FALSE]
    yb <- y[idx_boot]

    # CV LASSO on bootstrap sample
    cvfit <- cv.glmnet(
      x = Xb, y = yb,
      family = "binomial",
      alpha = alpha,
      penalty.factor = pf,
      standardize = TRUE,
      nfolds = nfolds,
      type.measure = "auc"
    )

    lam <- choose_lambda_by_protein_count(cvfit, prot_names = all_prots, cov_names = cov_names,
                                          min_prot = min_prot, max_prot = max_prot)
    lambda_used[b] <- lam

    beta <- as.matrix(coef(cvfit$glmnet.fit, s = lam))
    beta <- beta[, 1]
    names(beta) <- rownames(as.matrix(coef(cvfit$glmnet.fit, s = lam)))

    # selected proteins only
    sel_prot <- intersect(names(beta)[beta != 0], all_prots)
    sel_list[[b]] <- sel_prot
    coef_list[[b]] <- beta

    # OOB AUC (if enough OOB samples + both classes)
    if (length(idx_oob) >= 8 && length(unique(y[idx_oob])) == 2) {
      # fit final model on bootstrap sample at chosen lambda, predict on OOB
      fitb <- glmnet(
        x = Xb, y = yb,
        family = "binomial",
        alpha = alpha,
        penalty.factor = pf,
        standardize = TRUE,
        lambda = lam
      )
      p_oob <- as.numeric(predict(fitb, newx = X_full[idx_oob, , drop = FALSE], type = "response"))
      auc_oob[b] <- as.numeric(pROC::auc(pROC::roc(y[idx_oob], p_oob, quiet = TRUE)))
    }
  }

  # summarise selection frequency + sign stability
  sel_counts <- table(unlist(sel_list))
  freq <- as.numeric(sel_counts[all_prots])
  freq[is.na(freq)] <- 0
  freq <- freq / B

  # sign stability: among bootstraps where selected, fraction with positive coefficient
  sign_pos <- rep(NA_real_, length(all_prots))
  names(sign_pos) <- all_prots

  for (p in all_prots) {
    signs <- c()
    for (b in seq_len(B)) {
      beta <- coef_list[[b]]
      if (!is.null(beta) && p %in% names(beta) && beta[p] != 0) {
        signs <- c(signs, sign(beta[p]))
      }
    }
    if (length(signs) > 0) {
      sign_pos[p] <- mean(signs > 0)
    }
  }

  res <- data.frame(
    protein = all_prots,
    select_freq = freq,
    sign_pos_frac = sign_pos,
    stringsAsFactors = FALSE
  ) %>%
    arrange(desc(select_freq))

  list(
    summary = res,
    sel_list = sel_list,
    coef_list = coef_list,
    auc_oob = auc_oob,
    lambda_used = lambda_used,
    settings = list(B=B, include_age=include_age, include_sex=include_sex,
                    alpha=alpha, nfolds=nfolds, min_prot=min_prot, max_prot=max_prot)
  )
}

# Helper: Poland-based scaling for a set of proteins
get_scaler <- function(expr_mat, proteins){
  m <- rowMeans(expr_mat[proteins, , drop = FALSE], na.rm = TRUE)
  s <- apply(expr_mat[proteins, , drop = FALSE], 1, sd, na.rm = TRUE)
  s[s == 0 | is.na(s)] <- 1
  list(mu = m, sd = s)
}

apply_scaler <- function(expr_mat, proteins, scaler){
  Z <- expr_mat[proteins, , drop = FALSE]
  Z <- sweep(Z, 1, scaler$mu, "-")
  Z <- sweep(Z, 1, scaler$sd, "/")
  Z[is.na(Z)] <- 0
  Z
}

# ---- Option 2: refit logistic regression on Poland ----
fit_logistic_refit <- function(expr_mat, clin_df, proteins, cutoff = 0.32,
                               include_age = TRUE, include_sex = TRUE){
  # expr_mat: proteins x samples
  # clin_df order must match columns used
  df <- clin_df %>% mutate(y = ifelse(TAPSE_sPAP < cutoff, 1, 0))
  stopifnot(all(df$Barcode %in% colnames(expr_mat)))

  # Poland scaling inside this cohort (to be re-used later)
  scaler <- get_scaler(expr_mat[, df$Barcode, drop = FALSE], proteins)
  Z <- t(apply_scaler(expr_mat[, df$Barcode, drop = FALSE], proteins, scaler)) # samples x proteins
  Z <- as.data.frame(Z)
  Z$Barcode <- df$Barcode

  dd <- df %>% dplyr::select(Barcode, y, Age, Sex) %>% left_join(Z, by = "Barcode")

  # formula
  rhs <- c()
  if (include_age) rhs <- c(rhs, "Age")
  if (include_sex) rhs <- c(rhs, "Sex")
  rhs <- c(rhs, proteins)
  fml <- as.formula(paste("y ~", paste(rhs, collapse = " + ")))

  m <- glm(fml, data = dd, family = binomial())
  list(model = m, scaler = scaler, proteins = proteins,
       include_age = include_age, include_sex = include_sex)
}

predict_logistic_refit <- function(fit, expr_mat, clin_df){
  df <- clin_df
  stopifnot(all(df$Barcode %in% colnames(expr_mat)))

  Z <- t(apply_scaler(expr_mat[, df$Barcode, drop = FALSE], fit$proteins, fit$scaler))
  Z <- as.data.frame(Z)
  Z$Barcode <- df$Barcode

  dd <- df %>% dplyr::select(Barcode, Age, Sex, TAPSE_sPAP, TAPSE_group, Group) %>%
    left_join(Z, by = "Barcode")

  eta <- as.numeric(predict(fit$model, newdata = dd, type = "link"))
  p   <- as.numeric(predict(fit$model, newdata = dd, type = "response"))

  dd$score_eta <- eta
  dd$score_p   <- p
  dd
}

# ---- Option 3: signed mean z-score ----
signed_mean_score <- function(expr_mat, clin_df, proteins, sign_map, scaler){
  # sign_map: named vector (protein -> +1/-1)
  stopifnot(all(names(sign_map) %in% proteins))

  Z <- apply_scaler(expr_mat[, clin_df$Barcode, drop = FALSE], proteins, scaler) # proteins x samples
  s <- sign_map[proteins]
  Zs <- sweep(Z, 1, s, "*")
  score <- colMeans(Zs, na.rm = TRUE)

  out <- clin_df %>% dplyr::select(Barcode, Age, Sex, TAPSE_sPAP, TAPSE_group, Group)
  out$score_signed <- as.numeric(score[match(out$Barcode, names(score))])
  out
}

cor_spearman <- function(x, y) {
  ok <- is.finite(x) & is.finite(y)
  if (sum(ok) < 5) return(c(rho = NA_real_, p = NA_real_, n = sum(ok)))
  ct <- suppressWarnings(cor.test(x[ok], y[ok], method = "spearman", exact = FALSE))
  c(rho = unname(ct$estimate), p = ct$p.value, n = sum(ok))
}
